    _waitForFirstTick: function () {
       if (this._initialPlaytime == playtime) {
          setTimeout(function () { GeneralFunctions._waitForFirstTick() }, 1000);
       } else {
          this._ranFirstTick = true;
          this._runOnFirstTickList.forEach(func => {
             func();
          });
       }
