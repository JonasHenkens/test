/**
 * DataAccess.
 * @namespace
 */
var DataAccess = {

    _history_data_raw: null,
    _history_data: {},
    /**TODO */
    history_data_downloading: false,
    /**TODO */
    localStoragePrefix: 'redderke.',

    /**TODO */
    init: function () {
        let clearlist = this.get('DataAccessClearList')
        clearlist.forEach((x)=>{
            DataAccess.clear(x)
        })
    },

    /**TODO */
    set: function (key, value) {
        localStorage.setItem(this.localStoragePrefix + key, JSON.stringify(value));
    },

    /**TODO */
    get: function (key) {
        let localData = JSON.parse(localStorage.getItem(this.localStoragePrefix + key));
        if (localData !== null) {
            return localData
        } else {
            return typeof Constants.defaultValues[key] === 'undefined' ? null : Constants.defaultValues[key]
        }
    },

    /**TODO */
    clear: function (key) {
        localStorage.removeItem(key);
    },

    /**TODO */
    clearAll: function () {
        for (var key in localStorage) {
            if (key.startsWith(this.localStoragePrefix)) {
                localStorage.removeItem(key);
            }
        }
    },

    /**TODO */
    clearOnReload: function (key) {
        let clearlist = this.get('DataAccessClearList')
        clearlist.push(key);
        this.set('DataAccessClearList', clearlist)
    },

    /**TODO */
    getStrIdById: function (id) {
        for (const key in Constants.itemList) {
            if (Constants.itemList.hasOwnProperty(key) && Constants.itemList[key][0] == id) {
                return key;
            }
        }
    },

    /**TODO */
    getStrIdByName: function (name) {
        for (const key in Constants.itemList) {
            if (Constants.itemList.hasOwnProperty(key) && Constants.itemList[key][1] == name) {
                return key;
            }
        }
    },

    /**TODO */
    getApi: function (url, onload) {
        if (!url.startsWith('https')) {
            url = DataAccess.get('corsApiUrl') + url;
        }

        var xhr = new XMLHttpRequest();
        xhr.addEventListener("progress", () => console.log('progress'), false);
        xhr.addEventListener("load", () => onload(xhr), false);
        //xhr.addEventListener("error", transferFailed, false);
        //xhr.addEventListener("abort", transferCanceled, false);
        xhr.open('GET', url);
        xhr.send();
        return xhr;
    },

    /**TODO */
    getHistory: function (strId, onload = (() => { })) {
        if (DataAccess.get('MHVLocal')) {
            return this.get('MHVHistoryData')[strId];
        } else {
            if (typeof this._history_data[strId] === 'undefined') {
                if (this._history_data_raw === null) {
                    GeneralFunctions.updateHistoryData(() => {
                        this._generateHistoryCache(strId);
                        onload()
                    });
                } else {
                    this._generateHistoryCache(strId);
                }
            }
        }
        return this._history_data[strId];
    },

    /**TODO */
    getLastPrices: function (onload = (() => { })) {
        let lastPriceList = {}
        this.getHistory(Constants.marketList[0], onload)
        
        Constants.marketList.forEach(strId => {
            const priceList = this.getHistory(strId);
            if (typeof priceList !== 'undefined') {
                lastPriceList[strId] = priceList[priceList.length-1]
            }
        });
        return lastPriceList;
    },

    /**
     * TODO
     * @private
     */
    _generateHistoryCache: function (strId) {
        this._history_data[strId] = [];
        for (var key in this._history_data_raw) {
            if (this._history_data_raw.hasOwnProperty(key)) {
                var entry = this._history_data_raw[key];
                if (typeof entry !== 'undefined' && entry[0] == strId) {
                    this._history_data[strId].push(
                        [
                            entry[1],
                            entry[2],
                            Date.parse(entry[3] + ' CST')
                        ]
                    );
                }
            }
        }
    },
}
